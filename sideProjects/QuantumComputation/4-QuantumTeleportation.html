circuit = QuantumCircuit(3,3)

circuit.x(0) #Transfer to 1 state - NOT gate

circuit.barrier() #Barrier to make the circuit look better

circuit.h(1) # Apply H gate to q1
circuit.cx(1,2) # Apply CNOT to entangle q1 and q2

circuit.barrier()

circuit.draw(output='mpl')
